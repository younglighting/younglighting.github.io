    • random模块 文件开头导入模块import random，接着secret=random.randint(1,10)，引入1-10的随机整数。猜数字游戏代码块为下
import random
secret = random.randint(1,10)
print("*****i love fishc.com*****")
temp = input("猜个数字?")
guess = int(temp)
if guess == secret: 
         print ("猜对了")
else:
     if guess>secret:
               print("大大大了")
     else:
               print("小了小了")              
while guess!=secret:
     temp = input("不对哦，再试一次?")
     guess = int(temp)
     if guess == secret: 
         print ("猜对了")
     else:
          if guess>secret:
               print("大大大了")
          else:
               print("小了小了")
print("game over")
